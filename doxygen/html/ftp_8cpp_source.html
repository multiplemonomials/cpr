70 // ftpdata waits on a port at the local machine.
71 // When a connection is made, it receives a file from remote
72 // host if the ostream o is set, or it sends a file to the remote
73 // host if the istream i is set.
74 ftp::replycodea ftp::ftpbuf::ftpdata (int portno, istream* i, ostream* o,
75  const char* cmd, const char* arg)
76 {
77  sockinetbuf sb (sockbuf::sock_stream, 0);
78 
79  sb.bind_until_success (portno);
80  useraddr (sb.localaddr ());
81 
82  sb.listen (1);
83 
84  if (send_cmd (cmd, arg) >= ftp::rca_error)
85  return ftp::rca_error;
86 
87  if (o) {
88  sockbuf c = sb.accept ();
89 
90  // read data from c and put it in o
91  char buf [1024];
92  int rdsz;
93 
94  while ((rdsz = c.sgetn (buf, 1024)) != EOF)
95  o->write (buf, rdsz);
96  } else if (i) {
97  sockbuf c = sb.accept ();
98 
99  // read data from i and send it to c
100  char buf [1024];
101  int rdsz;
102  streambuf* rb = i->rdbuf ();
103 
104  while ((rdsz = rb->sgetn (buf, 1024)) > 0) {
105  int wrsz = c.sputn (buf, rdsz);
106  if (rdsz != wrsz)
107  cerr << "write error\n";
108  }
109  }
110 
111  // Note: sockbuf object c must have been destructed by the time you reach
112  // here.
113 
114  return get_response ();
115 }

117 ftp::replycodea ftp::ftpbuf::get_response ()
118  // get all the response that one can get and send all of them to o
119 {
120  // if o is 0, then we trash data.
121  int firstline = 1;
122  while (underflow () != EOF) {
123  int n = in_avail ();
124  if (n < 5)
125  continue;
126 
127  // data is of this form: 221 repsonse <CRLF> or 221-response <CRLF>
128  char* q = gptr ();
129  char* p = q;
130 
131  // zap upto <CRLF>
132  int i = 0;
133  for (i = 2; i <= n; i++, p++)
134  if (*p == '\r' && *(p+1) == '\n') {
135  break;
136  }
137  if (o)
138  o->write (q, i);
139  gbump (i);
140 
141  if (firstline) {
142  strncpy (replycode, q, 3);
143  replycode [3] = ' ';
144  if (q [3] == ' ')
145  break;
146  firstline = 0;
147  } else if (strncmp (q, replycode, 4) == 0)
148  break;
149  }
150 
151  return (replycodea) replycode [0];
152 }
