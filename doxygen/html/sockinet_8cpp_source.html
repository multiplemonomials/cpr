97 {
98  if (isdigit (*sn)) {

112 void sockinetaddr::setaddr(const char* host_name)
113 {
114  if ( (sin_addr.s_addr = inet_addr(host_name)) == INADDR_NONE)
115  {
116  hostent* hp = gethostbyname(host_name);
117 
118  if (hp == 0) throw cpr::exception (EADDRNOTAVAIL, "sockinetaddr::setaddr");
119 
120  memcpy(&sin_addr, hp->h_addr, hp->h_length);
121 
122  sin_family = hp->h_addrtype;
123  }
124  else

126  sin_family = sockinetbuf::af_inet;
