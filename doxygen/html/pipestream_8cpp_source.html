49 socketpp::sockbuf* detail::createpipestream(const char* cmd, int mode)
50 {
51  int sockets[2];
52  if (::socketpair(af_unix, socketpp::sockbuf::sock_stream, 0, sockets) == -1)
53  throw cpr::exception(errno, "cpr::pipestream()", "while invoking socketpair");
54 
55  pid_t pid = ::fork();
56  if (pid == -1) throw cpr::exception(errno, "cpr::pipestream()", "while invoking fork");
57 
58  if (pid == 0) {
59  // child process
60  if (::close(sockets[1]) == -1) throw cpr::exception(errno);
61 
62  //redirect stdout and stdin to the first socket
63  if ((mode & std::ios::in) && ::dup2(sockets[0], fileno(stdout)) == -1)
64  throw cpr::exception(errno);
65  if ((mode & std::ios::out) && ::dup2(sockets[0], fileno(stdin)) == -1)
66  throw cpr::exception(errno);
67 
68  //close the first socket, although the OS will keep piping stuff through it
69  if (::close(sockets[0]) == -1) throw cpr::exception(errno);
70 
71  const char* argv[4];
72  argv[0] = "/bin/sh";
73  argv[1] = "-c";
74  argv[2] = cmd;
75  argv[3] = '\0';
76  execve("/bin/sh", (char**) argv, environ);
77  _exit(0);
78  }
79 
80  // parent process
81  if (::close(sockets[0]) == -1) throw cpr::exception (errno);
82 
83  socketpp::sockbuf* s = new socketpp::sockbuf(socketpp::sockbuf::sockdesc(sockets[1]));
84 
85  if (!(mode & std::ios::out)) s->shutdown(socketpp::sockbuf::shut_write);
86  if (!(mode & std::ios::in)) s->shutdown(socketpp::sockbuf::shut_read);
87 
88  return s;
89 }
