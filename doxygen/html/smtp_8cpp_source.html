25  // get all the response that one can get and send all of them to output
26 {
27  // if output is 0, then we trash data.
28  while (underflow () != EOF) {
29  int n = in_avail ();
30  if (n < 5)
31  continue;

34  char* q = gptr ();

36 
37  // zap upto <CRLF>

39  if (*p == '\r' && *(p+1) == '\n') {
40  if (output)

42  gbump (i);

44  }
